539  virtual QCString trWriteList(int numEntries)
540  {
541  // this function is used to produce a comma-separated list of items.
542  // use generateMarker(i) to indicate where item i should be put.
543  QCString result;
544  int i;
545  // the inherits list contain `numEntries' classes
546  for (i=0;i<numEntries;i++)
547  {
548  // use generateMarker to generate placeholders for the class links!
549  result+=generateMarker(i); // generate marker for entry i in the list
550  // (order is left to right)
551 
552  if (i!=numEntries-1) // not the last entry, so we need a separator
553  {
554  if (i<numEntries-2) // not the fore last entry
555  result+=", ";
556  else // the fore last entry
557  result+=", och ";
558  }
559  }
560  return result;
561  }
