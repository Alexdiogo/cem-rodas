486  {
487  OutputGenerator *og=outputs->first();
488  while (og)
489  {
490  if (og->isEnabled()) (og->startPlainFile)(name);
491  og=outputs->next();
492  }
493  }
494  void endPlainFile()
495  {
496  OutputGenerator *og=outputs->first();
497  while (og)

499  if (og->isEnabled()) (og->endPlainFile)();
