651  for (i=0;i<numEntries;i++)
652  {

654  result+=generateMarker(i); // generate marker for entry i in the list
655  // (order is left to right)
656 
657  if (i!=numEntries-1) // not the last entry, so we need a separator

660  result+=", ";

662  result+=", and ";
